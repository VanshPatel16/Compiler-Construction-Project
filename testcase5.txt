    d4 <--- 12.34E+5 ;
    
    % SYNTACTICAL ERROR: Missing semicolon at the end of the statement
    c3 <--- d4 * 2.00

end

        % LEXICAL ERROR: The 23.abc trap. Lexer should throw TK_ERROR for 23. and TK_FIELDID for abc
        c3 <--- 23.abc ;
    endif
    
    while ( b2 < 20 )
        % LEXICAL ERROR: Unknown symbol '$' in the middle of an expression
        b2 <--- b2 + $ 1 ;
        
        % SYNTACTICAL ERROR: Missing 'endwhile' to close the loop
        
    % SYNTACTICAL ERROR: Reversed declaration syntax (should be type real : b5 ;)

    % LEXICAL ERROR: Invalid logical operator (&& instead of &&&)
    if ( ( b2 == 15 ) && ( c3 <= 10.00 ) ) then

    endif
